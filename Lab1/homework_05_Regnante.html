import numpy as np
import matplotlib.pyplot as plt
from scipy.constants import epsilon_0


k = 1.0 / (4.0 * np.pi * epsilon_0)

def potential_point_charge(x, y, Q=1e-9, x0=3.0, y0=-0.7, atol=1e-12):

    x = np.asarray(x)
    y = np.asarray(y)

    r = np.sqrt((x - x0)**2 + (y - y0)**2)

    V = k * Q / r
    V = np.where(np.isclose(r, 0.0, atol=atol), np.nan, V)
    return V

x = np.linspace(-5.0, 5.0, 201)
y = np.linspace(-5.0, 5.0, 201)
X, Y = np.meshgrid(x, y)

def drag_force(vx, vy, C):
    V_mag = np.sqrt(vx**2 + vy**2) #get V_mag first so then you can get its x and y components for drag force 
    Fx = -1 * (C * V_mag * vx) 
    Fy = -1 * (C * V_mag * vy)
    return Fx, Fy

vx = np.linspace(-4, 4, 20)
vy = np.linspace(-3, 3, 20)

VX, VY = np.meshgrid(vx, vy)

C = 1.0 #unity means set constant equal to 1 

FX, FY = drag_force(VX, VY, C)

fig, ax = plt.subplots(figsize=(7, 6))

ax.quiver(VX, VY, FX, FY, color='blue')

ax.set_xlabel("vx (m/s)")
ax.set_ylabel("vy (m/s)")
ax.set_title("Drag Force Vector Field", fontweight="bold")

plt.show()

import matplotlib.pyplot as plt
import numpy as np 

cylinder_radius = 2.0  # Cylinder radius
fig2, ax2 = plt.subplots(1,1)
disk = plt.Circle((0, 0), cylinder_radius, color='gray', alpha=0.2)
ax2.add_patch(disk)
ax2.set_xlim([-6,6])
ax2.set_ylim([-6,6])
ax2.set_aspect("equal")

u = 1.0                                               

# Make an (x,y) grid where we evaluate the flow field     # Grid approach mirrors Lecture 7 vector field setup (meshgrid)
x = np.linspace(-6, 6, 121)                               # Create x grid values (Lecture 7 uses np.linspace for grids)
y = np.linspace(-6, 6, 121)                               # Create y grid values (Lecture 7 uses np.linspace for grids)
X, Y = np.meshgrid(x, y)                                  # Make 2D coordinate arrays (Lecture 7: np.meshgrid)

#polar coordinate definitions
r = np.sqrt(X**2 + Y**2)                                 
theta = np.arctan2(Y, X)                                 

r_2_noerror = np.where(np.isclose(r, 0.0), np.nan, r**2)

# Polar components from the homework formula              # Vr and Vtheta equations are given in HW5 Q4
V_r = u * (1.0 - (cylinder_radius**2)/(r_2_noerror)) * np.cos(theta)          # Vr = u(1 - R^2/r^2)cos(theta)
V_theta = -u * (1.0 + (cylinder_radius**2)/(r_2_noerror)) * np.sin(theta)         # Vtheta = -u(1 + R^2/r^2)sin(theta)

# Convert polar velocity components to Cartesian          # HW5 Q4 gives Vx, Vy conversion formulas
Vx = V_r * np.cos(theta) - V_theta * np.sin(theta)             # Vx = Vr cosθ − Vθ sinθ
Vy = V_r * np.sin(theta) + V_theta * np.cos(theta)             # Vy = Vr sinθ + Vθ cosθ

# Handle the singularity at r=0 safely + enforce zero flow inside the cylinder
# (Because the formula has 1/r^2, and physically there is no flow inside the disk)
inside = r < cylinder_radius                              # Boolean mask for points inside the cylinder (HW5 Q4 requirement)
Vx = np.where(inside, 0.0, Vx)                            # Set Vx=0 inside cylinder (HW5 Q4 requirement)
Vy = np.where(inside, 0.0, Vy)                            # Set Vy=0 inside cylinder (HW5 Q4 requirement)

speed = np.sqrt(Vx**2 + Vy**2)                            # Speed magnitude |V| = sqrt(Vx^2+Vy^2) (HW5 Q4 requirement)
speed = np.where(inside, 0.0, speed)                      # Force speed=0 inside cylinder for coloring (HW5 Q4 requirement)

# Streamplot with color mapped to speed                   # Lecture 7 demonstrates ax.streamplot for vector fields
stream = ax2.streamplot(X, Y, Vx, Vy, color=speed,  cmap="plasma", density=2.0) 

# Colorbar in OO style                                    # Lecture 7 shows fig.colorbar(..., ax=...)
cbar = fig2.colorbar(stream.lines, ax=ax2)                # Attach colorbar to the line collection returned by streamplot
cbar.set_label("|V| (m/s)")                               # Label colorbar with units (HW5 Q4 requirement)

# Axis labels + title (OO style; no plt.plot usage)       # HW says label axes including units; Lecture 7 uses ax.set_... methods
ax2.set_xlabel("x (m)")                                   # x-axis label with meters (HW5 Q4)
ax2.set_ylabel("y (m)")                                   # y-axis label with meters (HW5 Q4)
ax2.set_title("Superfluid Flow Around a Cylinder (u = 1 m/s, R = 2 m)", fontweight="bold")  # Bold title (same style as Q3)

plt.show()                                                # Render the plot (Lecture 7 standard)
